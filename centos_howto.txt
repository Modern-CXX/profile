# CentOS 6 Gcc on devtoolset-8 = CentOS 8 Gcc on "Development Tools"
# no newer devtoolset for CentOS 8
# as Centos 8 is the newest OS and CentOS 9 is not released yet

# devtoolset (DTS) - newer version Development Tools
# from software collections (SCL)
$ sudo yum install centos-release-scl
$ sudo yum install devtoolset-8
$ vi ~/.bashrc
source scl_source enable devtoolset-8
$
# or,
$ scl enable devtoolset-8 bash
$

$ sudo blkid
/dev/sr0: UUID="2020-06-xxx" LABEL="CentOS-8-xxx" TYPE="iso9660"
$ sudo mkdir /mnt/data
$ sudo mount /dev/sr0 /mnt/data/

# mount on boot by name
$ sudo vi /etc/fstab
/dev/sr0            /mnt/data    iso9660    defaults    0    0
$
# or, mount on boot by UUID
$ sudo vi /etc/fstab
UUID=2020-06-xxx    /mnt/data    iso9660    defaults    0    0
$
