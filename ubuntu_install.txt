Ubuntu 20.10 install

Specify fast Mirror Address (change addr part before `/ubuntu/`):
https://mirrors.tuna.tsinghua.edu.cn/ubuntu/
# or:
# https://mirrors.aliyun.com/ubuntu/

$ sudo apt-get update
$ sudo apt-get upgrade

# gcc g++ c++
$ sudo apt install "build-essential"
$ sudo apt install "gdb"

# vi
$ sudo update-alternatives --config editor
$ sudo apt install "openssh-server"

$ sudo apt install samba
$ sudo smbpasswd -a ljh
$ vi /etc/samba/smb.conf
[homes]
   comment = Home Directories
   browseable = no
   read only = no
$

# firewall
$ sudo ufw app list
$ sudo ufw allow Samba

$ systemctl list-units | grep -i samba
$ systemctl status smbd
$ systemctl enable smbd nmbd
$ systemctl start smbd nmbd

$ timedatectl list-timezones
$ sudo timedatectl set-timezone "Asia/Shanghai"

$ sudo lsblk
$ sudo fdisk /dev/sdb
$ sudo mkfs.ext4 /dev/sdb1
$ sudo mkdir /mnt/data
$ sudo mount /dev/sdb1 /mnt/data
$ df -h
$ sudo vi /etc/fstab
  /dev/sdb1 /mnt/data ext4 defaults 0 0
$ sudo mount /mnt/data/
$

# find largest directory / file size
$ du -h -d1 /home/* | sort -hr | head -5

# dynamic network ip address
$ sudo vi /etc/netplan/00-installer-config.yaml
ethernets:
    eth0:
      dhcp4: true
$

# static network ip address, replace line `dhcp4:` above
$ sudo vi /etc/netplan/00-installer-config.yaml
ethernets:
    eth0:
      addresses:
        - 10.10.10.2/24
      gateway4: 10.10.10.1
      nameservers:
          search: [mydomain, otherdomain]
          addresses: [10.10.10.1, 1.1.1.1]
$ sudo netplan apply
